#include <iostream>
using namespace std;

class Shape{
public:
    double area;
    double volume;
};

class Cube : public Shape{
public:
    double side;

    Cube(double side){
        this->side = side;
        this->area = side * side * 6;
        this->volume = side * side * side;
    }
};

class Sphere : public Shape{
public:
    double radius;

    Sphere(double radius){
        this->radius = radius;
        this->area = 4 * 3.14159 * (radius * radius);
        this->volume = (4/3.0) * 3.14159 * (radius * radius * radius);
    }
};

int main() {
    Cube c(10);
       Sphere s(10);
cout << c.area << endl;
cout << c.volume << endl;
cout << s.area << endl;
cout << s.volume << endl;
    return 0;
}
